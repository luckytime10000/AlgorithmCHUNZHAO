class Solution(object):
    def moveZeroes(self, nums):
        """
        :type nums: List[int]
        :rtype: None Do not return anything, modify nums in-place instead.
        """
        index = 0
        for i in range(len(nums)):
            if nums[i] == 0:
                continue
            else:
                nums[index], nums[i] = nums[i], nums[index]
                index += 1
        
        return nums
