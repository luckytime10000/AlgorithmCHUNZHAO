# 1. 两数之和

class Solution(object):
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        result = []
        dic = {}
        for i in range(len(nums)):
            if target-nums[i] in dic:
                result = [dic[target-nums[i]], i]
                break
            else:
                dic[nums[i]] = i
        return result
